from flask import Flask, request
from twilio.twiml.voice_response import VoiceResponse

app = Flask(__name__)

@app.route("/process", methods=['POST'])
def process():
    speech = request.values.get('SpeechResult', '').lower()
    response = VoiceResponse()
    if "order" in speech:
        gather = response.gather(
            input='speech',
            action='/lookup-order',
            speechTimeout='auto'
        )
        gather.say("Sure, can you please provide your order number?")
    else:
        response.say("Sorry, I didn't understand. How can I help you?")
    return str(response)
